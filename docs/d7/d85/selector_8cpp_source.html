23void selector_screen::auton_ui_update(lv_event_t * e) {
24 lv_obj_t * tab1 = lv_event_get_current_target(e);
25 lv_obj_t * event_obj = lv_event_get_target(e);
26 lv_obj_t * autonLabel = lv_obj_get_child(tab1, 2);
27 lv_obj_t * allianceSwitch = lv_obj_get_child(tab1, 3);
28 lv_obj_t * skillSwitch = lv_obj_get_child(tab1, 4);
29 lv_obj_t * auton_dd = lv_obj_get_child(tab1, 5);
30
31 /* Autonomous selector follows this specification for updating the autonomous routine
32 * Skills - 0
33 * Red Alliance - 1, 2, 3, ... ,n
34 * Blue Alliance - -1, -2, -3, ... ,-n
35 */
36 if (event_obj == auton_dd) {
41 int currentAutonIndex = lv_dropdown_get_selected(auton_dd) + 1;
42 bool currentAlliance = lv_obj_has_state(allianceSwitch, LV_STATE_CHECKED);
43 int autonNum = currentAlliance ? currentAutonIndex * -1 : currentAutonIndex;
44 Autonomous::AutonSwitcher(autonNum);
45 }
46 /* Updates content of the dropdown list based on the alliance color, using
47 * the state of the alliance color switch, which is detected through "bubbling".
48 */
49 else if (event_obj == allianceSwitch) {
50 if(lv_obj_has_state(allianceSwitch, LV_STATE_CHECKED)) {
51 lv_dropdown_clear_options(auton_dd);
52 lv_dropdown_set_options(auton_dd, selector_screen::blueAutons);
53 lv_obj_set_style_border_color(auton_dd, lv_color_hex(0x0077c8), 0);
54 }
55 else {
56 lv_dropdown_clear_options(auton_dd);
57 lv_dropdown_set_options(auton_dd, selector_screen::redAutons);
58 lv_obj_set_style_border_color(auton_dd, lv_color_hex(0xd22730), 0);
59 }
60 // Switches the autonomous routine to the opposite alliance color, accounts for option reset
61 Autonomous::AutonSwitcher((Autonomous::auton > 0) ? -1 : 1);
62 }
63 else {
64 if (lv_obj_has_state(skillSwitch, LV_STATE_CHECKED)) {
65 // Remembers the last competition autonomous
66 selector_screen::lastAuton = Autonomous::auton;
67 selector_screen::lastColor = lv_obj_get_style_border_color(auton_dd, 0);
68
69 // Updates the autonomous routine to skills
70 Autonomous::AutonSwitcher(0);
71 lv_obj_add_state(allianceSwitch, LV_STATE_DISABLED);
72 lv_obj_add_state(auton_dd, LV_STATE_DISABLED);
73 lv_obj_set_style_border_color(auton_dd, lv_color_hex(0x7a7a7a), 0);
74 }
75 else {
76 Autonomous::AutonSwitcher(selector_screen::lastAuton);
77 lv_obj_clear_state(allianceSwitch, LV_STATE_DISABLED);
78 lv_obj_clear_state(auton_dd, LV_STATE_DISABLED);
79 lv_obj_set_style_border_color(auton_dd, selector_screen::lastColor, 0);
80 }
81 }
82 lv_label_set_text_fmt(autonLabel, "Current Auton: %s", Autonomous::autonName.c_str());
83}
