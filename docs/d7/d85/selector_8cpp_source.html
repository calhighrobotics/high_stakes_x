53static void auton_ui_update(lv_event_t * e) {
54 lv_event_code_t code = lv_event_get_code(e);
55 lv_obj_t * tab1 = lv_event_get_current_target(e);
56 lv_obj_t * auton_dd = lv_obj_get_child(tab1, -1);
57 lv_obj_t * skillSwitch = lv_obj_get_child(tab1, -2);
58 lv_obj_t * autonSwitch = lv_obj_get_child(tab1, -3);
59 lv_obj_t * autonLabel = lv_obj_get_child(tab1, -4);
60
61 /* Updates content of the dropdown list based on the alliance color, using
62 * the state of the alliance color switch, which is detected through "bubbling"
63 */
64 if (code == LV_EVENT_VALUE_CHANGED) {
65 if(lv_obj_has_state(autonSwitch, LV_STATE_CHECKED)) {
66 lv_dropdown_clear_options(auton_dd);
67 lv_dropdown_add_option(auton_dd, "Blue Left",0);
68 lv_dropdown_add_option(auton_dd, "Blue Right",1);
69 lv_obj_set_style_border_color(auton_dd, lv_color_hex(0x0077c8), 0);
70 }
71 else {
72 lv_dropdown_clear_options(auton_dd);
73 lv_dropdown_add_option(auton_dd, "Red Left",0);
74 lv_dropdown_add_option(auton_dd, "Red Right",1);
75 lv_obj_set_style_border_color(auton_dd, lv_color_hex(0xd22730), 0);
76 }
77 }
78
79 if (lv_obj_has_state(skillSwitch, LV_STATE_CHECKED)) {
80 lv_obj_add_state(autonSwitch, LV_STATE_DISABLED);
81 lv_obj_add_state(auton_dd, LV_STATE_DISABLED);
82 lv_obj_set_style_border_color(auton_dd, lv_color_hex(0x7a7a7a), 0);
83 }
84 else {
85 lv_obj_clear_state(autonSwitch, LV_STATE_DISABLED);
86 lv_obj_clear_state(auton_dd, LV_STATE_DISABLED);
87 }
88
89 auton_state_update(code, auton_dd, autonLabel, skillSwitch, autonSwitch);
90 //Updates the Autonomous routine based on the dropdown list and skills
91
