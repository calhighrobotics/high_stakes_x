42FMT_FUNC void format_error_code(detail::buffer<char>& out, int error_code, string_view message) noexcept {
43 // Report error code making sure that the output fits into
44 // inline_buffer_size to avoid dynamic memory allocation and potential
45 // bad_alloc.
46 out.try_resize(0);
47 static const char SEP[] = ": ";
48 static const char ERROR_STR[] = "error ";
49 // Subtract 2 to account for terminating null characters in SEP and ERROR_STR.
50 size_t error_code_size = sizeof(SEP) + sizeof(ERROR_STR) - 2;
51 auto abs_value = static_cast<uint32_or_64_or_128_t<int>>(error_code);
52 if (detail::is_negative(error_code)) {
53 abs_value = 0 - abs_value;
54 ++error_code_size;
55 }
56 error_code_size += detail::to_unsigned(detail::count_digits(abs_value));
57 auto it = buffer_appender<char>(out);
58 if (message.size() <= inline_buffer_size - error_code_size) format_to(it, FMT_STRING("{}{}"), message, SEP);
59 format_to(it, FMT_STRING("{}{}"), ERROR_STR, error_code);
60 FMT_ASSERT(out.size() <= inline_buffer_size, "");
61}

183// Replaces n by floor(n / pow(10, N)) returning true if and only if n is
184// divisible by pow(10, N).
185// Precondition: n <= pow(10, N + 1).
186template <int N> bool check_divisibility_and_divide_by_pow10(uint32_t& n) noexcept {
187 // The numbers below are chosen such that:
188 // 1. floor(n/d) = floor(nm / 2^k) where d=10 or d=100,
189 // 2. nm mod 2^k < m if and only if n is divisible by d,
190 // where m is magic_number, k is shift_amount
191 // and d is divisor.
192 //
193 // Item 1 is a common technique of replacing division by a constant with
194 // multiplication, see e.g. "Division by Invariant Integers Using
195 // Multiplication" by Granlund and Montgomery (1994). magic_number (m) is set
196 // to ceil(2^k/d) for large enough k.
197 // The idea for item 2 originates from Schubfach.
198 constexpr auto info = div_small_pow10_infos[N - 1];
199 FMT_ASSERT(n <= info.divisor * 10, "n is too large");
200 constexpr uint32_t magic_number = (1u << info.shift_amount) / info.divisor + 1;
201 n *= magic_number;
202 const uint32_t comparison_mask = (1u << info.shift_amount) - 1;
203 bool result = (n & comparison_mask) < magic_number;
204 n >>= info.shift_amount;
205 return result;
206}

208// Computes floor(n / pow(10, N)) for small n and N.
209// Precondition: n <= pow(10, N + 1).
210template <int N> uint32_t small_division_by_pow10(uint32_t n) noexcept {
211 constexpr auto info = div_small_pow10_infos[N - 1];
212 FMT_ASSERT(n <= info.divisor * 10, "n is too large");
213 constexpr uint32_t magic_number = (1u << info.shift_amount) / info.divisor + 1;
214 return (n * magic_number) >> info.shift_amount;
215}

989 int cache_index = (k - float_info<double>::min_k) / compression_ratio;
990 int kb = cache_index * compression_ratio + float_info<double>::min_k;
991 int offset = k - kb;
992
993 // Get base cache.
994 uint128_fallback base_cache = pow10_significands[cache_index];
995 if (offset == 0) return base_cache;
996
997 // Compute the required amount of bit-shift.
998 int alpha = floor_log2_pow10(kb + offset) - floor_log2_pow10(kb) - offset;
999 FMT_ASSERT(alpha > 0 && alpha < 64, "shifting error detected");
1000
1001 // Try to recover the real cache.
1002 uint64_t pow5 = powers_of_5_64[offset];
1003 uint128_fallback recovered_cache = umul128(base_cache.high(), pow5);
1004 uint128_fallback middle_low = umul128(base_cache.low(), pow5);
1005
1006 recovered_cache += middle_low.high();
1007
1008 uint64_t high_to_middle = recovered_cache.high() << (64 - alpha);
1009 uint64_t middle_to_low = recovered_cache.low() << (64 - alpha);
1010
1011 recovered_cache = uint128_fallback {(recovered_cache.low() >> alpha) | high_to_middle,
1012 ((middle_low.low() >> alpha) | middle_to_low)};
1013 FMT_ASSERT(recovered_cache.low() + 1 != 0, "");
1014 return {recovered_cache.high(), recovered_cache.low() + 1};
1015#endif
1016 }

1074// Remove trailing zeros from n and return the number of zeros removed (float)
1075FMT_INLINE int remove_trailing_zeros(uint32_t& n, int s = 0) noexcept {
1076 FMT_ASSERT(n != 0, "");
1077 // Modular inverse of 5 (mod 2^32): (mod_inv_5 * 5) mod 2^32 = 1.
1078 constexpr uint32_t mod_inv_5 = 0xcccccccd;
1079 constexpr uint32_t mod_inv_25 = 0xc28f5c29; // = mod_inv_5 * mod_inv_5
1080
1081 while (true) {
1082 auto q = rotr(n * mod_inv_25, 2);
1083 if (q > max_value<uint32_t>() / 100) break;
1084 n = q;
1085 s += 2;
1086 }
1087 auto q = rotr(n * mod_inv_5, 1);
1088 if (q <= max_value<uint32_t>() / 10) {
1089 n = q;
1090 s |= 1;
1091 }
1092 return s;
1093}

1095// Removes trailing zeros and returns the number of zeros removed (double)
1096FMT_INLINE int remove_trailing_zeros(uint64_t& n) noexcept {
1097 FMT_ASSERT(n != 0, "");
1098
1099 // This magic number is ceil(2^90 / 10^8).
1100 constexpr uint64_t magic_number = 12379400392853802749ull;
1101 auto nm = umul128(n, magic_number);
1102
1103 // Is n is divisible by 10^8?
1104 if ((nm.high() & ((1ull << (90 - 64)) - 1)) == 0 && nm.low() < magic_number) {
1105 // If yes, work with the quotient...
1106 auto n32 = static_cast<uint32_t>(nm.high() >> (90 - 64));
1107 // ... and use the 32 bit variant of the function
1108 int s = remove_trailing_zeros(n32, 8);
1109 n = n32;
1110 return s;
1111 }
1112
1113 // If n is not divisible by 10^8, work with n itself.
1114 constexpr uint64_t mod_inv_5 = 0xcccccccccccccccd;
1115 constexpr uint64_t mod_inv_25 = 0x8f5c28f5c28f5c29; // = mod_inv_5 * mod_inv_5
1116
1117 int s = 0;
1118 while (true) {
1119 auto q = rotr(n * mod_inv_25, 2);
1120 if (q > max_value<uint64_t>() / 100) break;
1121 n = q;
1122 s += 2;
1123 }
1124 auto q = rotr(n * mod_inv_5, 1);
1125 if (q <= max_value<uint64_t>() / 10) {
1126 n = q;
1127 s |= 1;
1128 }
1129
1130 return s;
1131}

1133// The main algorithm for shorter interval case
1134template <typename T> FMT_INLINE decimal_fp<T> shorter_interval_case(int exponent) noexcept {
1135 decimal_fp<T> ret_value;
1136 // Compute k and beta
1137 const int minus_k = floor_log10_pow2_minus_log10_4_over_3(exponent);
1138 const int beta = exponent + floor_log2_pow10(-minus_k);
1139
1140 // Compute xi and zi
1141 using cache_entry_type = typename cache_accessor<T>::cache_entry_type;
1142 const cache_entry_type cache = cache_accessor<T>::get_cached_power(-minus_k);
1143
1144 auto xi = cache_accessor<T>::compute_left_endpoint_for_shorter_interval_case(cache, beta);
1145 auto zi = cache_accessor<T>::compute_right_endpoint_for_shorter_interval_case(cache, beta);
1146
1147 // If the left endpoint is not an integer, increase it
1148 if (!is_left_endpoint_integer_shorter_interval<T>(exponent)) ++xi;
1149
1150 // Try bigger divisor
1151 ret_value.significand = zi / 10;
1152
1153 // If succeed, remove trailing zeros if necessary and return
1154 if (ret_value.significand * 10 >= xi) {
1155 ret_value.exponent = minus_k + 1;
1156 ret_value.exponent += remove_trailing_zeros(ret_value.significand);
1157 return ret_value;
1158 }
1159
1160 // Otherwise, compute the round-up of y
1161 ret_value.significand = cache_accessor<T>::compute_round_up_for_shorter_interval_case(cache, beta);
1162 ret_value.exponent = minus_k;
1163
1164 // When tie occurs, choose one of them according to the rule
1165 if (exponent >= float_info<T>::shorter_interval_tie_lower_threshold &&
1166 exponent <= float_info<T>::shorter_interval_tie_upper_threshold) {
1167 ret_value.significand = ret_value.significand % 2 == 0 ? ret_value.significand : ret_value.significand - 1;
1168 } else if (ret_value.significand < xi) {
1169 ++ret_value.significand;
1170 }
1171 return ret_value;
1172}

1174template <typename T> decimal_fp<T> to_decimal(T x) noexcept {
1175 // Step 1: integer promotion & Schubfach multiplier calculation.
1176
1177 using carrier_uint = typename float_info<T>::carrier_uint;
1178 using cache_entry_type = typename cache_accessor<T>::cache_entry_type;
1179 auto br = bit_cast<carrier_uint>(x);
1180
1181 // Extract significand bits and exponent bits.
1182 const carrier_uint significand_mask = (static_cast<carrier_uint>(1) << num_significand_bits<T>()) - 1;
1183 carrier_uint significand = (br & significand_mask);
1184 int exponent = static_cast<int>((br & exponent_mask<T>()) >> num_significand_bits<T>());
1185
1186 if (exponent != 0) { // Check if normal.
1187 exponent -= exponent_bias<T>() + num_significand_bits<T>();
1188
1189 // Shorter interval case; proceed like Schubfach.
1190 // In fact, when exponent == 1 and significand == 0, the interval is
1191 // regular. However, it can be shown that the end-results are anyway same.
1192 if (significand == 0) return shorter_interval_case<T>(exponent);
1193
1194 significand |= (static_cast<carrier_uint>(1) << num_significand_bits<T>());
1195 } else {
1196 // Subnormal case; the interval is always regular.
1197 if (significand == 0) return {0, 0};
1198 exponent = std::numeric_limits<T>::min_exponent - num_significand_bits<T>() - 1;
1199 }
1200
1201 const bool include_left_endpoint = (significand % 2 == 0);
1202 const bool include_right_endpoint = include_left_endpoint;
1203
1204 // Compute k and beta.
1205 const int minus_k = floor_log10_pow2(exponent) - float_info<T>::kappa;
1206 const cache_entry_type cache = cache_accessor<T>::get_cached_power(-minus_k);
1207 const int beta = exponent + floor_log2_pow10(-minus_k);
1208
1209 // Compute zi and deltai.
1210 // 10^kappa <= deltai < 10^(kappa + 1)
1211 const uint32_t deltai = cache_accessor<T>::compute_delta(cache, beta);
1212 const carrier_uint two_fc = significand << 1;
1213
1214 // For the case of binary32, the result of integer check is not correct for
1215 // 29711844 * 2^-82
1216 // = 6.1442653300000000008655037797566933477355632930994033813476... * 10^-18
1217 // and 29711844 * 2^-81
1218 // = 1.2288530660000000001731007559513386695471126586198806762695... * 10^-17,
1219 // and they are the unique counterexamples. However, since 29711844 is even,
1220 // this does not cause any problem for the endpoints calculations; it can only
1221 // cause a problem when we need to perform integer check for the center.
1222 // Fortunately, with these inputs, that branch is never executed, so we are
1223 // fine.
1224 const typename cache_accessor<T>::compute_mul_result z_mul =
1225 cache_accessor<T>::compute_mul((two_fc | 1) << beta, cache);
1226
1227 // Step 2: Try larger divisor; remove trailing zeros if necessary.
1228
1229 // Using an upper bound on zi, we might be able to optimize the division
1230 // better than the compiler; we are computing zi / big_divisor here.
1231 decimal_fp<T> ret_value;
1232 ret_value.significand = divide_by_10_to_kappa_plus_1(z_mul.result);
1233 uint32_t r = static_cast<uint32_t>(z_mul.result - float_info<T>::big_divisor * ret_value.significand);
1234
1235 if (r < deltai) {
1236 // Exclude the right endpoint if necessary.
1237 if (r == 0 && (z_mul.is_integer & !include_right_endpoint)) {
1238 --ret_value.significand;
1239 r = float_info<T>::big_divisor;
1240 goto small_divisor_case_label;
1241 }
1242 } else if (r > deltai) {
1243 goto small_divisor_case_label;
1244 } else {
1245 // r == deltai; compare fractional parts.
1246 const typename cache_accessor<T>::compute_mul_parity_result x_mul =
1247 cache_accessor<T>::compute_mul_parity(two_fc - 1, cache, beta);
1248
1249 if (!(x_mul.parity | (x_mul.is_integer & include_left_endpoint))) goto small_divisor_case_label;
1250 }
1251 ret_value.exponent = minus_k + float_info<T>::kappa + 1;
1252
1253 // We may need to remove trailing zeros.
1254 ret_value.exponent += remove_trailing_zeros(ret_value.significand);
1255 return ret_value;
1256
1257 // Step 3: Find the significand with the smaller divisor.
1258
1259small_divisor_case_label:
1260 ret_value.significand *= 10;
1261 ret_value.exponent = minus_k + float_info<T>::kappa;
1262
1263 uint32_t dist = r - (deltai / 2) + (float_info<T>::small_divisor / 2);
1264 const bool approx_y_parity = ((dist ^ (float_info<T>::small_divisor / 2)) & 1) != 0;
1265
1266 // Is dist divisible by 10^kappa?
1267 const bool divisible_by_small_divisor = check_divisibility_and_divide_by_pow10<float_info<T>::kappa>(dist);
1268
1269 // Add dist / 10^kappa to the significand.
1270 ret_value.significand += dist;
1271
1272 if (!divisible_by_small_divisor) return ret_value;
1273
1274 // Check z^(f) >= epsilon^(f).
1275 // We have either yi == zi - epsiloni or yi == (zi - epsiloni) - 1,
1276 // where yi == zi - epsiloni if and only if z^(f) >= epsilon^(f).
1277 // Since there are only 2 possibilities, we only need to care about the
1278 // parity. Also, zi and r should have the same parity since the divisor
1279 // is an even number.
1280 const auto y_mul = cache_accessor<T>::compute_mul_parity(two_fc, cache, beta);
1281
1282 // If z^(f) >= epsilon^(f), we might have a tie when z^(f) == epsilon^(f),
1283 // or equivalently, when y is an integer.
1284 if (y_mul.parity != approx_y_parity) --ret_value.significand;
1285 else if (y_mul.is_integer & (ret_value.significand % 2 != 0)) --ret_value.significand;
1286 return ret_value;
1287}
