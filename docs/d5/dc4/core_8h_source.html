424 FMT_CONSTEXPR_CHAR_TRAITS bool starts_with(basic_string_view<Char> sv) const noexcept {
425 return size_ >= sv.size_ && std::char_traits<Char>::compare(data_, sv.data_, sv.size_) == 0;
426 }
427

429 return size_ >= 1 && std::char_traits<Char>::eq(*data_, c);

435
436 // Lexicographically compare this string reference to other.
437 FMT_CONSTEXPR_CHAR_TRAITS auto compare(basic_string_view other) const -> int {
438 size_t str_size = size_ < other.size_ ? size_ : other.size_;
439 int result = std::char_traits<Char>::compare(data_, other.data_, str_size);
440 if (result == 0) result = size_ == other.size_ ? 0 : (size_ < other.size_ ? -1 : 1);
441 return result;
442 }
443
444 FMT_CONSTEXPR_CHAR_TRAITS friend auto operator==(basic_string_view lhs, basic_string_view rhs) -> bool {
445 return lhs.compare(rhs) == 0;
446 }
447
448 friend auto operator!=(basic_string_view lhs, basic_string_view rhs) -> bool { return lhs.compare(rhs) != 0; }
449
450 friend auto operator<(basic_string_view lhs, basic_string_view rhs) -> bool { return lhs.compare(rhs) < 0; }
451
452 friend auto operator<=(basic_string_view lhs, basic_string_view rhs) -> bool { return lhs.compare(rhs) <= 0; }
453
454 friend auto operator>(basic_string_view lhs, basic_string_view rhs) -> bool { return lhs.compare(rhs) > 0; }
455
456 friend auto operator>=(basic_string_view lhs, basic_string_view rhs) -> bool { return lhs.compare(rhs) >= 0; }

844// A buffer that writes to an output iterator when flushed.
845template <typename OutputIt, typename T, typename Traits = buffer_traits>
846class iterator_buffer final : public Traits, public buffer<T> {
847 private:
848 OutputIt out_;
849
850 enum { buffer_size = 256 };
851
852 T data_[buffer_size];
853 protected:
854 FMT_CONSTEXPR20 void grow(size_t) override {
855 if (this->size() == buffer_size) flush();
856 }
857

859 auto size = this->size();
860 this->clear();
861 out_ = copy_str<T>(data_, data_ + this->limit(size), out_);
862 }
863 public:

865 : Traits(n), buffer<T>(data_, 0, buffer_size), out_(out) {}

868

870

872 flush();
873 return out_;
874 }
875

1969// Parses the range [begin, end) as an unsigned integer. This function assumes
1970// that the range is non-empty and the first character is a digit.
1971template <typename Char>
1972FMT_CONSTEXPR auto parse_nonnegative_int(const Char*& begin, const Char* end, int error_value) noexcept -> int {
1973 FMT_ASSERT(begin != end && '0' <= *begin && *begin <= '9', "");
1974 unsigned value = 0, prev = 0;
1975 auto p = begin;
1976 do {
1977 prev = value;
1978 value = value * 10 + unsigned(*p - '0');
1979 ++p;
1980 } while (p != end && '0' <= *p && *p <= '9');
1981 auto num_digits = p - begin;
1982 begin = p;
1983 if (num_digits <= std::numeric_limits<int>::digits10) return static_cast<int>(value);
1984 // Check for overflow.
1985 const unsigned max = to_unsigned((std::numeric_limits<int>::max)());
1986 return num_digits == std::numeric_limits<int>::digits10 + 1 && prev * 10ull + unsigned(p[-1] - '0') <= max
1987 ? static_cast<int>(value)
1988 : error_value;
1989}

2004template <typename Char, typename Handler>
2005FMT_CONSTEXPR auto do_parse_arg_id(const Char* begin, const Char* end, Handler&& handler) -> const Char* {
2006 Char c = *begin;
2007 if (c >= '0' && c <= '9') {
2008 int index = 0;
2009 constexpr int max = (std::numeric_limits<int>::max)();
2010 if (c != '0') index = parse_nonnegative_int(begin, end, max);
2011 else ++begin;
2012 if (begin == end || (*begin != '}' && *begin != ':')) throw_format_error("invalid format string");
2013 else handler.on_index(index);
2014 return begin;
2015 }
2016 if (!is_name_start(c)) {
2017 throw_format_error("invalid format string");
2018 return begin;
2019 }
2020 auto it = begin;
2021 do { ++it; } while (it != end && (is_name_start(*it) || ('0' <= *it && *it <= '9')));
2022 handler.on_name({begin, to_unsigned(it - begin)});
2023 return it;
2024}
