29class dynamic_arg_list {
30 // Workaround for clang's -Wweak-vtables. Unlike for regular classes, for
31 // templates it doesn't complain about inability to deduce single translation
32 // unit for placing vtable. So storage_node_base is made a fake template.
33 template <typename = void> struct node {
34 virtual ~node() = default;
35 std::unique_ptr<node<>> next;
36 };
37
38 template <typename T> struct typed_node : node<> {
39 T value;
40
41 template <typename Arg> FMT_CONSTEXPR typed_node(const Arg& arg) : value(arg) {}
42
43 template <typename Char> FMT_CONSTEXPR typed_node(const basic_string_view<Char>& arg)
44 : value(arg.data(), arg.size()) {}
45 };
46
47 std::unique_ptr<node<>> head_;
48 public:
49 template <typename T, typename Arg> const T& push(const Arg& arg) {
50 auto new_node = std::unique_ptr<typed_node<T>>(new typed_node<T>(arg));
51 auto& value = new_node->value;
52 new_node->next = std::move(head_);
53 head_ = std::move(new_node);
54 return value;
55 }
56};
57} // namespace detail

78 template <typename T> struct need_copy {
79 static constexpr detail::type mapped_type = detail::mapped_type_constant<T, Context>::value;
80
81 enum {
82 value = !(detail::is_reference_wrapper<T>::value ||
83 std::is_same<T, basic_string_view<char_type>>::value ||
84 std::is_same<T, detail::std_string_view<char_type>>::value ||
85 (mapped_type != detail::type::cstring_type && mapped_type != detail::type::string_type &&
86 mapped_type != detail::type::custom_type))
87 };
88 };

90 template <typename T> using stored_type =
91 conditional_t<std::is_convertible<T, std::basic_string<char_type>>::value &&
92 !detail::is_reference_wrapper<T>::value,
93 std::basic_string<char_type>, T>;
94
95 // Storage of basic_format_arg must be contiguous.
96 std::vector<basic_format_arg<Context>> data_;
97 std::vector<detail::named_arg_info<char_type>> named_info_;
98
99 // Storage of arguments not fitting into basic_format_arg must grow
100 // without relocation because items in data_ refer to it.
101 detail::dynamic_arg_list dynamic_args_;
102
103 friend class basic_format_args<Context>;

105 unsigned long long get_types() const {
106 return detail::is_unpacked_bit | data_.size() |
107 (named_info_.empty() ? 0ULL : static_cast<unsigned long long>(detail::has_named_args_bit));
108 }

110 const basic_format_arg<Context>* data() const { return named_info_.empty() ? data_.data() : data_.data() + 1; }

112 template <typename T> void emplace_arg(const T& arg) { data_.emplace_back(detail::make_arg<Context>(arg)); }
113
114 template <typename T> void emplace_arg(const detail::named_arg<char_type, T>& arg) {
115 if (named_info_.empty()) {
116 constexpr const detail::named_arg_info<char_type>* zero_ptr {nullptr};
117 data_.insert(data_.begin(), {zero_ptr, 0});
118 }
119 data_.emplace_back(detail::make_arg<Context>(detail::unwrap(arg.value)));
120 auto pop_one = [](std::vector<basic_format_arg<Context>>* data) { data->pop_back(); };
121 std::unique_ptr<std::vector<basic_format_arg<Context>>, decltype(pop_one)> guard {&data_, pop_one};
122 named_info_.push_back({arg.name, static_cast<int>(data_.size() - 2u)});
123 data_[0].value_.named_args = {named_info_.data(), named_info_.size()};
124 guard.release();
125 }
126 public:
127 constexpr dynamic_format_arg_store() = default;

146 template <typename T> void push_back(const T& arg) {
147 if (detail::const_check(need_copy<T>::value)) emplace_arg(dynamic_args_.push<stored_type<T>>(arg));
148 else emplace_arg(detail::unwrap(arg));
149 }

166 template <typename T> void push_back(std::reference_wrapper<T> arg) {
167 static_assert(need_copy<T>::value, "objects of built-in types and string views are always copied");
168 emplace_arg(arg.get());
169 }
170
176 template <typename T> void push_back(const detail::named_arg<char_type, T>& arg) {
177 const char_type* arg_name = dynamic_args_.push<std::basic_string<char_type>>(arg.name).c_str();
178 if (detail::const_check(need_copy<T>::value)) {
179 emplace_arg(fmt::arg(arg_name, dynamic_args_.push<stored_type<T>>(arg.value)));
180 } else {
181 emplace_arg(fmt::arg(arg_name, arg.value));
182 }
183 }
184
186 void clear() {
187 data_.clear();
188 named_info_.clear();
189 dynamic_args_ = detail::dynamic_arg_list();
190 }
191
198 void reserve(size_t new_cap, size_t new_cap_named) {
199 FMT_ASSERT(new_cap >= new_cap_named, "Set of arguments includes set of named arguments");
200 data_.reserve(new_cap);
201 named_info_.reserve(new_cap_named);
202 }
203};
